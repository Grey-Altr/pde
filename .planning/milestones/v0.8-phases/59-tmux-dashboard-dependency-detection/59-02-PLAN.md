Create `bin/pane-agent-activity.sh` (chmod +x):
```bash
#!/usr/bin/env bash
# pane-agent-activity.sh — streams agent spawn/complete events from NDJSON
# TMUX-02: Agent activity pane shows real-time agent spawn/complete events

NDJSON="${1:-}"
if [ -z "$NDJSON" ]; then
  echo "Usage: pane-agent-activity.sh <ndjson-path>"
  exit 1
fi

echo "[ agent activity ] waiting for session events..."
echo ""

tail -F "${NDJSON}" 2>/dev/null | while IFS= read -r line; do
  event_type=$(echo "$line" | jq -r '.event_type // empty' 2>/dev/null)
  case "$event_type" in
    subagent_start)
      ts=$(echo "$line" | jq -r '.ts | split("T")[1] | split(".")[0]' 2>/dev/null)
      agent=$(echo "$line" | jq -r '.agent_type // "agent"' 2>/dev/null)
      printf '\033[32m[%s] SPAWN  %s\033[0m\n' "$ts" "$agent"
      ;;
    subagent_stop)
      ts=$(echo "$line" | jq -r '.ts | split("T")[1] | split(".")[0]' 2>/dev/null)
      agent=$(echo "$line" | jq -r '.agent_type // "agent"' 2>/dev/null)
      printf '\033[33m[%s] DONE   %s\033[0m\n' "$ts" "$agent"
      ;;
  esac
done
```
- Green (`\033[32m`) for SPAWN, Yellow (`\033[33m`) for DONE
- Uses `jq -r` for raw output (not `-C`), explicit ANSI codes via printf
- `tail -F` (uppercase) for self-healing file follow
- `2>/dev/null` on tail to suppress "file not found" noise before file exists

Create `bin/pane-file-changes.sh` (chmod +x):
```bash
#!/usr/bin/env bash
# pane-file-changes.sh — streams file creation/modification events from NDJSON
# TMUX-04: File changes pane shows files created/modified in real-time

NDJSON="${1:-}"
if [ -z "$NDJSON" ]; then
  echo "Usage: pane-file-changes.sh <ndjson-path>"
  exit 1
fi

echo "[ file changes ] waiting for session events..."
echo ""

tail -F "${NDJSON}" 2>/dev/null | while IFS= read -r line; do
  event_type=$(echo "$line" | jq -r '.event_type // empty' 2>/dev/null)
  case "$event_type" in
    tool_use)
      tool=$(echo "$line" | jq -r '.tool_name // empty' 2>/dev/null)
      fpath=$(echo "$line" | jq -r '.file_path // empty' 2>/dev/null)
      ts=$(echo "$line" | jq -r '.ts | split("T")[1] | split(".")[0]' 2>/dev/null)
      if [ -n "$fpath" ]; then
        case "$tool" in
          Write)
            printf '\033[32m[%s] CREATE %s\033[0m\n' "$ts" "$fpath"
            ;;
          Edit)
            printf '\033[33m[%s] MODIFY %s\033[0m\n' "$ts" "$fpath"
            ;;
          *)
            printf '\033[34m[%s] %-6s %s\033[0m\n' "$ts" "$tool" "$fpath"
            ;;
        esac
      fi
      ;;
  esac
done
```
- Green for CREATE (Write tool), Yellow for MODIFY (Edit tool), Blue for other tools with file paths
- Only prints lines where `file_path` is non-empty (Bash tool calls without file paths are skipped)
- Timestamp extracted with same jq pattern as other pane scripts

Create `bin/pane-log-stream.sh` (chmod +x):
```bash
#!/usr/bin/env bash
# pane-log-stream.sh — streams all events as a colored log
# TMUX-05: Log stream pane shows structured event log with severity filtering

NDJSON="${1:-}"
if [ -z "$NDJSON" ]; then
  echo "Usage: pane-log-stream.sh <ndjson-path>"
  exit 1
fi

echo "[ log stream ] waiting for session events..."
echo ""

tail -F "${NDJSON}" 2>/dev/null | while IFS= read -r line; do
  event_type=$(echo "$line" | jq -r '.event_type // empty' 2>/dev/null)
  ts=$(echo "$line" | jq -r '.ts | split("T")[1] | split(".")[0]' 2>/dev/null)

  # Skip empty/unparseable lines
  [ -z "$event_type" ] && continue

  # Color by event severity/category
  case "$event_type" in
    session_start|session_end)
      # White bold — lifecycle events
      printf '\033[1;37m[%s] %s\033[0m\n' "$ts" "$event_type"
      ;;
    subagent_start|subagent_stop)
      agent=$(echo "$line" | jq -r '.agent_type // ""' 2>/dev/null)
      printf '\033[36m[%s] %-20s %s\033[0m\n' "$ts" "$event_type" "$agent"
      ;;
    tool_use)
      tool=$(echo "$line" | jq -r '.tool_name // ""' 2>/dev/null)
      printf '\033[90m[%s] %-20s %s\033[0m\n' "$ts" "$event_type" "$tool"
      ;;
    phase_started|phase_complete|wave_started|wave_complete|plan_started|plan_complete)
      printf '\033[35m[%s] %s\033[0m\n' "$ts" "$event_type"
      ;;
    *)
      # Default gray for unknown event types
      printf '\033[90m[%s] %s\033[0m\n' "$ts" "$event_type"
      ;;
  esac
done
```
- White bold for session lifecycle, Cyan for agents, Gray for tool_use (high frequency, low priority), Magenta for pipeline events
- Shows ALL events (no filtering) — this is the "full log" pane
- Unknown event types still display (gray) — forward-compatible
